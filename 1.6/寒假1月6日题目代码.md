#include<iostream>
#include<algorithm>
#include<string>
#include<vector>
using namespace std;

const int N = 1010;
vector<int> a[N];

vector<int> add(vector<int> A, vector<int> B)
{
	int t = 0;
	vector<int> C;
	for (int i = 0; i < A.size() || i < B.size(); i ++)
	{
		if (i < A.size())	t += A[i];
		if (i < B.size())	t += B[i];
		C.push_back(t % 10);
		t /= 10;
	}
	if (t)	C.push_back(1);
	return C;
}

void init()
{
	a[1].push_back(1);
	a[2].push_back(2);
	for (int i = 3; i <= 1000; i ++)
	{
		a[i] = add(a[i - 1], a[i - 2]);
	}
}

void print(vector<int> A)
{
	for (int i = A.size() - 1; i >= 0; i --)	printf("%d", A[i]);
}

int main(void)
{
	init();
	
	int m, n;
	cin >> m >> n;
	print(a[n - m]);
	
	return 0;
}

typedef struct Fri{
	int h, w, id;
}Fri;

Fri mans[N];
int ans[N];

bool cmp(Fri a, Fri b)
{
	if (a.h != b.h)	return a.h < b.h;
	else	return a.w < b.w;
}

int main(void)
{
	int t;
	scanf("%d", &t);
	while (t --)
	{
		int n;
		scanf("%d", &n);
		
		memset(ans, -1, sizeof(ans));
		for (int i = 1; i <= n; i ++)
		{
			int a, b;
			scanf("%d%d", &a, &b);
			mans[i] = {a, -b, i};
			mans[i + n] = {b, -a, i};
		}
		sort(mans + 1, mans + 2 * n + 1, cmp);
		
		int e = 0, m = INT_MAX;
		for (int i = 1; i <= 2 * n; i ++)
		{
			int h = mans[i].h, w = -mans[i].w, id = mans[i].id;
			if (w > m)	ans[id] = e;
			else if (w < m)
			{
				m = w;
				e = id;
			}
		}
		
		for (int i = 1; i <= n; i ++)	printf("%d ", ans[i]);
