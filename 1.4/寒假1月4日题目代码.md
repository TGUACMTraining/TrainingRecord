```cpp
#include<cstdio>

using namespace std;

typedef long long LL;

const int mod = 1e9 + 7;
const int N = 100010;
const int M = 5010;

int a[N];
int b[M];
int c[M][3];

void init()
{
	for (int i = 0; i < M; i ++)
		for (int j = 0; j <= 2; j ++)
		if (!j)	c[i][j] = 1;
		else	c[i][j] = (c[i - 1][j] + c[i - 1][j - 1]) % mod;
}

int main(void)
{
	int n;
	init();
	int res = 0;
	cin >> n;
	for (int i = 0; i < n; i ++)
	{
		scanf("%d", &a[i]);
		b[a[i]] ++;
	}
	
	for (int i = 1; i < M; i ++)
	{
		if (b[i])
		{
			b[i] --;
			for (int j = i; j < M; j ++)
			{
				if (b[j] && i + j < M && b[i + j] >= 2)
				{
					if (i != j)
					{
						res = (res + (LL)(b[i] + 1) * b[j] * c[b[i + j]][2] % mod) % mod; 
					}
					else
					{
						res = (res + (LL)c[b[i] + 1][2] * c[b[i + j]][2] % mod) % mod;
					}
				}
			}
			b[i] ++;
		}
	}
	
	cout << res << endl;
	
	return 0;
}
```

```cpp
#include<cstdio>
#include<iostream>
#include<algorithm>
#include<cmath>
#include<climits>
using namespace std;

typedef long long LL;
typedef pair<LL, LL> PII;

int n;
PII ans;
int res = INT_MAX;
PII a[12];

void dfs(int x, int n)
{
	if (x == n)	return;
	for (int i = x; i < n; i ++)
	{
		LL t = ans.first;
		ans.first *= a[i].first;
		ans.second += a[i].second;
		res = min((LL)res, abs(ans.second - ans.first));
		dfs(i + 1, n);
		ans.first = t;
		ans.second -= a[i].second;
	}
	
}

int main(void)
{
	int n;
	cin >> n;
	for (int i = 0; i < n; i ++)
	{
		int s, b;
		cin >> s >> b;
		a[i] = {s, b};
	}
	
	
	for (int i = 0; i < n; i ++)
	{
		ans = {1, 0};
		dfs(i, n);
	}
	cout << res << endl;
	
	return 0;
}
```

```cpp
	{
		int w, h, n;
		scanf("%d%d%d", &w, &h, &n);
		int answ = 1, ansh = 1;

			answ *= 2;

		{
			ansh *= 2;
			h /= 2;
		}
		ll res = answ * ansh;

	}
	
	return 0;
```

## CODEFORCES 1472D. Even-Odd Game
```cpp
#include<cstdio>
#include<iostream>
#include<algorithm>
using namespace std;

typedef long long ll;

const int N = 200010;
int a[N];

bool cmp(int a, int b)
{
	return a > b;
}

int main(void)
{
	int t;
	cin >> t;
	while (t --)
	{
		int n;
		scanf("%d", &n);
		for (int i = 0; i < n; i ++)
		{
			scanf("%d", &a[i]);
		}
		sort(a, a + n, cmp);
		
		ll bob = 0, alice = 0;
		for (int i = 0; i < n; i ++)
		{
			if (i % 2 && a[i] % 2)	bob += a[i];
			if (!(i % 2) && !(a[i] % 2)) alice += a[i];
		}
		
		if (bob > alice)	puts("Bob");
		else if (alice > bob)	puts("Alice");
		else	puts("Tie");
	}
	
	return 0;
}
```
