```cpp
	{

		t /= 10;
```

```cpp
#include<cstdio>
#include<iostream>
#include<algorithm>
using namespace std;

typedef long long LL;

const int N = 23;
LL w[N][N][N];

LL dfs(int a, int b, int c)
{
	if (w[a][b][c])	return w[a][b][c];
	if (a == 0 || b == 0 || c == 0)	return 1;
	if (a < b && b < c)	w[a][b][c] = dfs(a, b, c - 1) + dfs(a, b - 1, c - 1) - dfs(a, b - 1, c);
	else w[a][b][c] = dfs(a - 1, b, c) + dfs(a - 1, b - 1, c) + dfs(a - 1, b, c - 1) - dfs(a - 1, b - 1, c - 1);
	return w[a][b][c];
}

int main(void)
{
	for (int i = 0; i <= 20; i ++)
		for (int j = 0; j <= 20; j ++)
			for (int k = 0; k <= 20; k ++)
				w[i][j][k] = dfs(i, j, k);
	
	while (1)
	{
		LL a, b, c;
		scanf("%lld%lld%lld", &a, &b, &c);
		if (a == -1 && b == -1 && c == -1)	break;
		else if (a <= 0 || b <= 0 || c <= 0)	printf("w(%lld, %lld, %lld) = %lld\n", a, b, c, w[0][0][0]);
		else if (a > 20 || b > 20 || c > 20)	printf("w(%lld, %lld, %lld) = %lld\n", a, b, c, w[20][20][20]);
		else	printf("w(%lld, %lld, %lld) = %lld\n", a, b, c, w[a][b][c]);
	}
	
	return 0;
}
```

```cpp
	int num = 0;
	int k;
	for (k = 1; s[i + k] >= '0' && s[i + k] <= '9'; k ++)
	{
		num *= 10;
		num += s[i + k] - '0';
	}
	string res;
	while (num --)
	{
		for (int j = i + k; s[j] != ']'; j ++)
		{
			if (s[j] == '[')
			{
				int left = 1;
				res += dfs(j);
				while (left)
				{
					j ++;
					if (s[j] == '[')	left ++;
					if (s[j] == ']')	left --;
				}
			}
			else	res += s[j];
		}
	}
	return res;
}

		if (!left)
```

## Codeforces Round #694 (Div. 2) C. Strange Birthday Party
```cpp
#include<cstdio>
#include<iostream>
#include<algorithm>
using namespace std;
typedef long long ll;

const int N = 300010;
int a[N], b[N];

bool cmp(int a, int b)
{
	return a > b;
}

int main(void)
{
	int t;
	scanf("%d", &t);
	while (t --)
	{
		int n, m;
		scanf("%d%d", &n, &m);
		for (int i = 0; i < n; i ++)	scanf("%d", &a[i]);
		for (int i = 0; i < m; i ++)	scanf("%d", &b[i]);
		
		sort(a, a + n, cmp);
		
		ll ans = 0;
		int cnt = 0;
		for (int i = 0; i < n; i ++)
		{
			if (b[a[i] - 1] > b[cnt])	ans += b[cnt ++];
			else	ans += b[a[i] - 1];
		}
		
		printf("%lld\n", ans);
	}
	
	return 0;
}
```
