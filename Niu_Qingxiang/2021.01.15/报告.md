for (ll i=2;i<=n;i++) {
		if (!notp[i]) {
			pri[++pri[0]]=i;
		}
		for (ll j=1;j<=pri[0] && i*pri[j]<=n;j++) {
			notp[pri[j]*i]=1;
			if (i%pri[j]==0) {
				break;
			}
		}
	}
