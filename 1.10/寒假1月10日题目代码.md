## AcWing 900. 整数划分
### 暴力做法 三重循环 O(n^3)
```cpp
#include<cstdio>
#include<iostream>
#include<algorithm>
using namespace std;

const int N = 1010, mod = 1e9 + 7;
int n;
int f[N][N];

void print()
{
    for (int i = 0; i <= n; i ++)
    {
        for (int j = 0; j <= n; j ++)   printf("%d ", f[i][j]);
        puts("");
    }
    puts("");
}

int main(void)
{
    cin >> n;
    
    for (int i = 0; i <= n; i ++)   f[i][0] = 1;
    
    for (int i = 1; i <= n; i ++)
    {
        for (int j = 1; j <= n; j ++)
        {
            if (j < i)  f[i][j] = f[i - 1][j];
            else
                for (int k = 0; i * k <= j; k ++)
                    f[i][j] = (f[i][j] + f[i - 1][j - i * k]) % mod;
        }
        
    }
    
    
    
    cout << f[n][n] << endl;
    
    return 0;
}
```

### 时间复杂度优化, 二重循环 O(n^2)
```cpp

#include<iostream>
#include<algorithm>
using namespace std;

const int N = 1010, mod = 1e9 + 7;
int n;
int f[N][N];

    {
        for (int j = 0; j <= n; j ++)   printf("%d ", f[i][j]);
        puts("");
    }
    puts("");
}

int main(void)
{
    cin >> n;
    
    for (int i = 0; i <= n; i ++)   f[i][0] = 1;
    
    for (int i = 1; i <= n; i ++)
    {
        for (int j = 1; j <= n; j ++)
        {
            if (j < i)  f[i][j] = f[i - 1][j];
            else    f[i][j] = (f[i - 1][j] + f[i][j - i]) % mod;
        }
    }
    
    
    
    cout << f[n][n] << endl;
    
    return 0;
}
```

### 时间, 空间复杂度优化, 二重循环一位数组, O(n^2)
```cpp
#include<cstdio>
#include<iostream>
#include<algorithm>
using namespace std;

const int N = 1010, mod = 1e9 + 7;
int n;
int f[N];

int main(void)
{
    cin >> n;
    
    f[0] = 1;
    for (int i = 1; i <= n; i ++)
        for (int j = i; j <= n; j ++)
            f[j] = (f[j] + f[j - i]) % mod;
    
    
    
    cout << f[n] << endl;
    
    return 0;
}
```

## AcWing 285. 没有上司的舞会
```cpp
#include<cstdio>
#include<iostream>
#include<algorithm>
#include<cstring>
using namespace std;

const int N = 6010;

int n;
int h[N], e[N], ne[N], idx;
int happy[N];
int f[N][2];
bool fa[N];

void add(int a, int b)
{
    e[idx] = b, ne[idx] = h[a], h[a] = idx ++;
}

void dfs(int u)
{
    f[u][1] = happy[u];
    
    for (int i = h[u]; i != -1; i = ne[i])
    {
        int j = e[i];
        dfs(j);
        
        f[u][1] += f[j][0];
        f[u][0] += max(f[j][0], f[j][1]);
    }
}

int main(void)
{
    cin >> n;
    
    for (int i = 1; i <= n; i ++)   scanf("%d", &happy[i]);
    
    memset(h, -1, sizeof(h));
    for (int i = 0; i < n - 1; i ++)
    {
        int a, b;
        scanf("%d%d", &a, &b);
        add(b, a);
        fa[a] = true;
    }
    
    int root = 1;
    while (fa[root])    root ++;
    
    dfs(root);
    
    printf("%d\n", max(f[root][0], f[root][1]));
    
    return 0;
}
```

## AcWing 901. 滑雪
```cpp
#include<cstdio>
#include<iostream>
#include<algorithm>
#include<cstring>
using namespace std;

const int N = 310;

int n, m;
int g[N][N];
int f[N][N];

int dirc[5] = {-1, 0, 1, 0, -1};

bool check(int x, int y)
{
    if (x >= 1 && x <= n && y >= 1 && y <= m)   return true;
    return false;
}

int dp(int x, int y)
{
    int &v = f[x][y];
    if (v != -1)    return v;
    
    v = 1;
    for (int i = 0; i < 4; i ++)
    {
        int a = x + dirc[i], b = y + dirc[i + 1];
        if (check(a, b) && g[x][y] > g[a][b])
            v = max(v, dp(a, b) + 1);
    }
    
    return v;
}

int main(void)
{
    scanf("%d%d", &n, &m);
    for (int i = 1; i <= n; i ++)
        for (int j = 1; j <= m; j ++)
            scanf("%d", &g[i][j]);
            
    memset(f, -1, sizeof(f));
    
    int res = 0;
    for (int i = 1; i <= n; i ++)
        for (int j = 1; j <= m; j ++)
            res = max(res, dp(i, j));
    
    printf("%d\n", res);
    
    return 0;
}
```
